354  LoomModule* Create(JsonVariant module) const
355  {
356  const char* name = module["name"];
357 
358  for (auto elem : factory::to_array(LookupTable) ) {
359  if ( strcmp(name, elem.name) == 0 ) {
360 
361  if (module["params"].is<JsonArray>()) {
362  // Generate according to list of parameters
363  LPrintln("[Factory] Creating: ", name);
364  return elem.Construct(module["params"]);
365  } else if ( module["params"].is<const char*>() && strcmp(module["params"], "default") == 0 ) {
366  // Generate using default parameters
367  if (elem.ConstructDefault == nullptr) {
368  LPrintln("[Factory] ", "No default constructor for module '", name, "'");
369  return nullptr;
370  } else {
371  return elem.ConstructDefault();
372  }
373  } else {
374  // Invalid parameters
375  LPrintln("[Factory] ", "Invalid parameters in module '", name, "'");
376  return nullptr;
377  }
378  }
379  }
380 
381  LPrintln("'", name, "' could not be created");
382  return nullptr;
383  }
